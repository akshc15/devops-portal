app/screener.py — the AI brain
import os
from openai import OpenAI
import PyPDF2
import io, json
client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))
def extract_text_from_pdf(pdf_bytes: bytes) -> str:
reader = PyPDF2.PdfReader(io.BytesIO(pdf_bytes))
return " ".join(page.extract_text() for page in reader.pages)
def analyze_resume(resume_text: str, job_role: str = "DevOps Engineer") -> dict:
prompt = f"""
You are an expert resume reviewer and career coach.
Analyze this resume for the role: {job_role}
RESUME:
{resume_text[:3000]}
Return ONLY valid JSON (no markdown) with this exact structure:
{{
"score": 0-100,
"grade": "A/B/C/D/F",
"strengths": ["strength1", "strength2"],
"weaknesses": ["weakness1", "weakness2"],
"missing_keywords": ["keyword1", "keyword2"],
"improved_summary": "rewritten professional summary",
"ats_tips": ["tip1", "tip2", "tip3"]
}}
"""
response = client.chat.completions.create(
model="gpt-4o-mini", # cheapest + fast
messages=[{"role": "user", "content": prompt}],
temperature=0.3
)
return json.loads(response.choices[0].message.content)
app/main.py — the API
from fastapi import FastAPI, UploadFile, File, Form, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from prometheus_fastapi_instrumentator import Instrumentator
from app.screener import extract_text_from_pdf, analyze_resume
import time
app = FastAPI(title="AI Resume Screener", version="1.0.0")
app.add_middleware(CORSMiddleware, allow_origins=["*"],
allow_methods=["*"], allow_headers=["*"])
Instrumentator().instrument(app).expose(app)
@app.get("/health")
def health(): return {"status": "ok", "service": "ai-resume-screener"}
@app.post("/analyze")
async def analyze(
resume: UploadFile = File(...),
job_role: str = Form(default="DevOps Engineer")
):
if not resume.filename.endswith(".pdf"):
raise HTTPException(400, "Only PDF files accepted")
pdf_bytes = await resume.read()
if len(pdf_bytes) > 5 * 1024 * 1024: # 5MB limit
raise HTTPException(400, "File too large. Max 5MB.")
start = time.time()
text = extract_text_from_pdf(pdf_bytes)
if len(text.strip()) < 100:
raise HTTPException(400, "Could not extract text from PDF")
result = analyze_resume(text, job_role)
result["processing_time"] = round(time.time() - start, 2)
result["job_role"] = job_role
return result
@app.post("/rewrite")
async def rewrite(resume: UploadFile = File(...), job_role: str = Form(...)):
pdf_bytes = await resume.read()
text = extract_text_from_pdf(pdf_bytes)
result = analyze_resume(text, job_role)
return {"original_score": result["score"], "improved_summary": result["improved_summary"],
"ats_tips": result["ats_tips"], "missing_keywords": result["missing_keywords"]}

app/k8s_client.py — gather cluster context
from kubernetes import client, config
import requests, os
def get_k8s_context(namespace="default") -> dict:
"""Gather cluster state for AI analysis"""
try:
config.load_incluster_config() # when running inside K8s
except:
config.load_kube_config() # when running locally
v1 = client.CoreV1Api()
apps_v1 = client.AppsV1Api()
# Get all pod statuses
pods = v1.list_namespaced_pod(namespace)
pod_summary = []
for pod in pods.items:
status = pod.status.phase
restarts = sum(c.restart_count for c in (pod.status.container_statuses or []))
pod_summary.append({
"name": pod.metadata.name,
"status": status,
"restarts": restarts,
"ready": all(c.ready for c in (pod.status.container_statuses or []))
})
# Get recent events (errors show up here)
events = v1.list_namespaced_event(namespace)
warning_events = [
{"type": e.type, "reason": e.reason, "message": e.message, "object": e.involved_object.name}
for e in events.items if e.type == "Warning"
][-10:] # last 10 warnings
return {"pods": pod_summary, "events": warning_events, "namespace": namespace}
def get_pod_logs(pod_name: str, namespace="default", lines=50) -> str:
config.load_incluster_config() if os.getenv("KUBERNETES_SERVICE_HOST") else config.load_kube_config()
v1 = client.CoreV1Api()
try:
return v1.read_namespaced_pod_log(pod_name, namespace, tail_lines=lines)
except Exception as e:
return f"Could not get logs: {e}"
app/main.py — the AI assistant
from fastapi import FastAPI
from pydantic import BaseModel
from openai import OpenAI
from app.k8s_client import get_k8s_context, get_pod_logs
from prometheus_fastapi_instrumentator import Instrumentator
import os, json
app = FastAPI(title="K8s AI Assistant")
Instrumentator().instrument(app).expose(app)
ai = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))
class Question(BaseModel):
question: str
namespace: str = "default"
pod_name: str = None
@app.get("/health")
def health(): return {"status": "ok"}
@app.post("/ask")
def ask_cluster(q: Question):
# Gather real cluster context
context = get_k8s_context(q.namespace)
logs = ""
if q.pod_name:
logs = get_pod_logs(q.pod_name, q.namespace)
prompt = f"""You are an expert Kubernetes SRE. Analyze this cluster state and answer the question.
CLUSTER STATE:
{json.dumps(context, indent=2)}
POD LOGS (last 50 lines):
{logs}
QUESTION: {q.question}
Respond with:
1. ROOT CAUSE (1-2 sentences)
2. EXACT FIX COMMANDS (kubectl commands to run right now)
3. PREVENTION (how to avoid this next time)
"""
response = ai.chat.completions.create(
model="gpt-4o-mini",
messages=[{"role": "user", "content": prompt}]
)
return {
"question": q.question,
"cluster_summary": {"total_pods": len(context["pods"]),
"unhealthy": sum(1 for p in context["pods"] if not p["ready"]),
"warnings": len(context["events"])},
"ai_analysis": response.choices[0].message.content
}
STEP 2Deploy to K8s — With Cluster Read Access›

This app needs to read from the Kubernetes API. We use a ServiceAccount with read-only permissions.

k8s/rbac.yaml — read-only cluster access
# ServiceAccount for the app
apiVersion: v1
kind: ServiceAccount
metadata:
name: k8s-ai-assistant
---
# ClusterRole — read-only access to pods, events, logs
apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRole
metadata:
name: k8s-ai-reader
rules:
- apiGroups: [""]
resources: ["pods", "pods/log", "events", "nodes", "services"]
verbs: ["get", "list", "watch"] # read only — no create/delete
- apiGroups: ["apps"]
resources: ["deployments", "replicasets"]
verbs: ["get", "list", "watch"]
---
apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRoleBinding
metadata:
name: k8s-ai-reader-binding
subjects:
- kind: ServiceAccount
name: k8s-ai-assistant
namespace: default
roleRef:
kind: ClusterRole
name: k8s-ai-reader
apiGroup: rbac.authorization.k8s.io
